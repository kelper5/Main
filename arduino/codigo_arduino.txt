const int PINO_SENSOR_TEMPERATURA = A2;
const int PINO_SENSOR_LDR = A5;

float temperaturaCelsius;
int valorLuminosidade;

void setup() {
  Serial.begin(9600);
}

void loop() {
  float valorLeitura = analogRead(PINO_SENSOR_TEMPERATURA);
  valorLuminosidade = analogRead(PINO_SENSOR_LDR);

  temperaturaCelsius = (valorLeitura * 5.0 / 1023.0) / 0.01;

  float tensao = valorLuminosidade * (5.0 / 1023.0);

  if (tensao <= 0.01) tensao = 0.01;

  float resistenciaLDR = (5.0 - tensao) * 10000.0 / tensao;

  float lux = pow( (10000.0 / resistenciaLDR), 1.4 ) * 100;

  Serial.print(temperaturaCelsius);
  Serial.print(";");
  Serial.println(lux);

  delay(2000);
}
